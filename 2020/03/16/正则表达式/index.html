正则表达式

正则表达式是用于匹配字符串中字符组合的模式，在JavaScript中，正则表达式也是对象

作用：

  1. 验证表单
  2. 过滤页面敏感词并且替换它
  3. 获取指定字符串部分

正则表达式特点：

  1. 灵活 逻辑性和功能性非常强
  2. 可以迅速极简单的方式达到字符串的复杂控制
  3. 实际开发中，一般是直接复制写好的正则表达式，但是要学会根据实际情况去更改表达式

正则表达式在JavaScript中的使用

  1. 通过调用RegExp对象的构造函数创建

    var regexp=new RegExp(/123/);
输入的必须是123;

  2. 利用字面量创建

    var 变量名=/表达式/;
var arr=/123/

测试正则表达式text

test()正则对象方法，用于检测字符串是否符合该规则，该对象会返回true或false，其参数是测试字符串

regexobj.test(str)
// regexobj是写的正则表达式
// str我们要测试的文本
// 检测str文本是否符合我们写的正则表达式规范

var a = /123456/;
console.log(a.test(123)); //false
console.log(a.test(123456));//true

正则表达式的组成

正则表达式可以由简单的字符构成，比如/abc/

也可以是简单和特殊字符的组合 比如/ab*c/，特殊字符被称为元字符，有特殊意义的专用符号如 ^ $ +等

正则表达式里面是不需要加引号，无论是数字还是字符串

精确匹配

边界符 说明
^ 表示匹配行首的文本（以谁开头）
$ 表示匹配行尾的文本（以谁结尾） 
var rg=/abc/; 只要是含abc这个字符串返回的都是true
console.log(rg.test('abc'))  //true
console.log(rg.test('abcd'))  //true
console.log(rg.test('aabcd'))  //true


var reg=/^abc/;只要abc开头都是true
console.log(reg.test('abc'))  //true
console.log(reg.test('abcd'))  //true
console.log(reg.test('aabcd'))  //false


var rr=/^abc$/; 精确匹配  必须是abc字符串才符合规范
console.log(rr.test('abc'))  //true
console.log(rr.test('abcd'))  //false
console.log(rr.test('aabcd'))  //false
console.log(rr.test('abcabc'))  //false

字符类

【】表示有一系列字符可供选择，只要匹配其中一个就可以

var rg=/[abc]/ 只要包含a或者bh或者c任何一个都返回true
console.log(rg.test('andy') //true
console.log(rg.test('baby') //true
console.log(rg.test('color') //true
console.log(rr.test('red') //false

var rg=/[^abc$]/ 三选一，只有是a或者是b后者是c这三个字母才返回true
console.log(rg.test('aa') //false
console.log(rg.test('a') //true
console.log(rg.test('b') //true
console.log(rg.test('c') //true

【-】方括号 内部范围符

var rg=/^[a-z]$/  26个字母返回任何一个都是true
console.log(rg.test('a') //true
console.log(rg.test('A') //false
console.log(rg.test('1') //false
console.log(rg.test('z') //true

字符组合

var reg=/^[a-zA-Z0-9]$/26个英文字母（大小写都可以）任何一个字符都返回true
console.log(reg.test('z') //true
console.log(reg.test('B') //true  
console.log(reg.test('_') //true
console.log(reg.test('!') //false

如果【】里面有^ 表示取反的意思 

var reg=/^[^a-zA-Z0-9]$/26个英文字母（大小写都可以）任何一个字符都返回true
console.log(reg.test('z') //false
console.log(reg.test('B') //false  
console.log(reg.test('_') //false
console.log(reg.test('!') //true

量词符

用来设定某个模式出现的次数

简单理解就是让下面的a这个字符重复多少次

var reg=/^a*$/  *相当于>=0可以出现0次或者很多次
console.log（reg.test('')) //true
console.log（reg.test('a')) //true
console.log（reg.test('aa')) //true
var reg=/^a+$/  *相当于>=1至少出现1次或者很多次
console.log（reg.test('')) //false
console.log（reg.test('a')) //true
console.log（reg.test('aa')) //true
var reg=/^a?$/  *相当于1||0才返回true
console.log（reg.test('')) //true
console.log（reg.test('a')) //true
console.log（reg.test('aa')) //false
var reg=/^a{3}$/  *相当于重复3次才返回true
console.log（reg.test('')) //false
console.log（reg.test('a')) //false
console.log（reg.test('aa')) //false
console.log（reg.test('aaa')) //false
var reg=/^a{3,}$/  *相当于>=3
console.log（reg.test('')) //false
console.log（reg.test('a')) //false
console.log（reg.test('aaa')) //true
console.log（reg.test('aaaa')) //true
var reg=/^a{3,16}$/  *相当于>=3并且<=16
console.log（reg.test('')) //false
console.log（reg.test('a')) //false
console.log（reg.test('aa')) //false
console.log（reg.test('aaaaa')) //true

正则表达式的替换

replace（）方法可以实现替换字符操作，用来替换的可以是一个字符串或是一个正则表达式

var str='andy和red'
var newStr=str.replace('andy','baby')
console.log(newStr) // baby和red

obj.replace(参数1，参数2，参数3)
obj.replace(/激情|gay/g，"**")
参数1 ：字符串或者正则表达式
参数2 ：修饰符 按照什么样的模式来匹配 
//g全局  i忽略大小写 gi全局匹配+忽略大小写
参数3 :替换的字符

var str='andy和red'
var newStr=str.replace(/andy/,'baby')
console.log(newStr) // baby和red
